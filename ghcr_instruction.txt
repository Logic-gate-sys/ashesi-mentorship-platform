# LOGIN
docker login --username logic-gate-sys --password PAT_TOKEN ghcr.io

 # Tag an already built image 
docker tag mentor-app-mentor-app ghcr.io/logic-gate-sys/ashesi-mentorship-platform:latest

 # BUILD DOCKER IMAGE
 docker build -t ghcr.io/logic-gate-sys/mentor-app:v1

 # PUSH BUILT IMAGE TO GHCR

 # :::::::::::::::::::USING COMPOSE::::::::::::::::

     image: ghcr.io/logic-gate-sys/mentor-app:v1 
     dockerfile: Dockerfile
